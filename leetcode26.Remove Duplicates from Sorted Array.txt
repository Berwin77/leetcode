class Solution:
    def removeElement(self, nums, val):
        """
        :type nums: List[int]
        :type val: int
        :rtype: int
        """
        index = 0
        while index != len(nums):
            if nums[index] == val:
                nums.pop(index)
            else:
                index += 1
            print(nums, index)
        return len(nums)

class Solution:
    def removeDuplicates(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        s = set(nums)
        return list(s)
